  <script>
    Telegram.WebApp.ready();

    const products = [
      { id: 1, name: "Губка для посуды", price: 5000, category: "Хозтовары" },
      { id: 2, name: "Моющее средство", price: 12000, category: "Хозтовары" },
      { id: 3, name: "Цемент 50 кг", price: 65000, category: "Стройматериалы" },
      { id: 4, name: "Молоток", price: 35000, category: "Инструменты" },
    ];

    const catalog = document.getElementById("catalog");

    const categories = [...new Set(products.map(p => p.category))];

    categories.forEach(cat => {
      const title = document.createElement("div");
      title.className = "category";
      title.innerText = cat;
      catalog.appendChild(title);

      products
        .filter(p => p.category === cat)
        .forEach(p => {
          const card = document.createElement("div");
          card.className = "card";
          card.innerHTML = `
            <div>${p.name}</div>
            <div class="price">${p.price.toLocaleString()} сум</div>
            <button onclick="addToCart('${p.name}')">➕ В корзину</button>
          `;
          catalog.appendChild(card);
        });
    });

    function addToCart(name) {
      Telegram.WebApp.showAlert(`Товар "${name}" добавлен в корзину`);
    }
  </script>
